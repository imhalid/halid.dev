```js showLineNumbers title="useRule.jsx"
const useRule = (input) => {
  const rule = new RegExp(/([-0-9]{2,3})\*\.([0-9]{6})+/g);

  if (input.match(rule)) {
    return input.match(rule);
  } else {
    return console.log("No match");
  }
};
```
